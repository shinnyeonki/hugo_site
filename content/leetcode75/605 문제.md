```cpp
class Solution {
public:
    bool canPlaceFlowers(vector<int>& flowerbed, int n) {
        if (n > placeFolwers(flowerbed)) return false;
        else return true;
    }

    int placeFolwers(vector<int>& flowerbed){
        int result = 0;
        for (int i = 0 ; i < flowerbed.size() ; i++){
            int temp = i;
            while(flowerbed[i] == 0){
                i++;
            }
            result += (int)((i - temp - 1) / 2);
        }
        return result;
    }
};
```
