TABLE length(rows) AS "File Count"
FROM ""
FLATTEN file.tags AS Tag
WHERE Tag 
	AND !contains(file.path, "05.clipping") 
	AND !contains(file.path, "89.Obsidian")
GROUP BY Tag
SORT length(rows) desc
