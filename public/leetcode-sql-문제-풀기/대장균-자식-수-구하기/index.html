join 방식으로 풀기

SELECT a.id, COUNT(b.parent_id) AS child_count
FROM ecoli_data a LEFT JOIN ecoli_data b ON a.ID = b.parent_id
GROUP BY a.ID
ORDER BY a.id

subquery

SELECT 
    E.ID,
    (
        SELECT COUNT(*)
        FROM ECOLI_DATA AS ED
        WHERE ED.PARENT_ID = E.ID
    ) AS CHILD_COUNT
FROM 
    ECOLI_DATA E
ORDER BY 
    E.ID;
